The jingle cursor trail needs a small script. Just copy and paste it into the header or footer directive in your Bear settings:

<script>
(function () {
  const reduceMotion = window.matchMedia("(prefers-reduced-motion: reduce)");
  if (reduceMotion.matches) return;

  const darkMQ = window.matchMedia("(prefers-color-scheme: dark)");
  const isDark = () => darkMQ.matches;

  const EMOJIS = ["‚ú®", "‚≠êÔ∏è"];
  let lastSpark = 0;

  function spawnSparkle(x, y) {
    const s = document.createElement("span");
    s.className = "bearry-sparkle is-on";
    s.textContent = EMOJIS[(Math.random() * EMOJIS.length) | 0];

    const scale = 0.9 + Math.random() * 0.2;
    s.style.transform = `translate(-50%, -50%) scale(${scale})`;
    s.style.left = x + "px";
    s.style.top = y + "px";
    s.style.opacity = isDark() ? "0.85" : "0.5";

    document.body.appendChild(s);
    setTimeout(() => s.remove(), 800);
  }

  function onMouseMove(e) {
    const now = performance.now();
    if (now - lastSpark < 55) return;
    lastSpark = now;
    spawnSparkle(e.clientX, e.clientY);
  }

  window.addEventListener("mousemove", onMouseMove);
})();
</script>
